async function cadastrar() { const nome = document.getElementById('nomeUsuario').value; if (!nome) return alert("Por favor, digite um nome!"); document.getElementById('status').innerText = "Analisando biometria..."; const detec = await faceapi.detectSingleFace('video', new faceapi.TinyFaceDetectorOptions()).withFaceLandmarks().withFaceDescriptor(); if (!detec) return alert("Rosto não detectado."); try { const res = await fetch(URL, { headers: { "X-Master-Key": API_KEY } }); const base = await res.json(); let lista = (base.record && base.record.users) ? base.record.users : []; // --- INÍCIO DA CORREÇÃO TÉCNICA --- if (lista.length > 0) { // Reconvertemos os dados do JSON para Float32Array (essencial para a comparação) const labeledDescriptors = lista.map(u => { const descritosTratados = new Float32Array(u.descriptor); return new faceapi.LabeledFaceDescriptors(u.nome, [descritosTratados]); }); const faceMatcher = new faceapi.FaceMatcher(labeledDescriptors, 0.5); // 0.5 é mais rigoroso que 0.6 const match = faceMatcher.findBestMatch(detec.descriptor); if (match.label !== 'unknown') { document.getElementById('status').innerText = "Pronto"; return alert(`Atenção: Este rosto já está cadastrado como "${match.label}".`); } } // --- FIM DA CORREÇÃO --- lista.push({ nome, descriptor: Array.from(detec.descriptor) }); await fetch(URL, { method: 'PUT', headers: { "Content-Type": "application/json", "X-Master-Key": API_KEY }, body: JSON.stringify({ users: lista }) }); alert(`Sucesso! ${nome} cadastrado.`); } catch (e) { alert("Erro de rede: " + e); } document.getElementById('status').innerText = "Pronto"; }
